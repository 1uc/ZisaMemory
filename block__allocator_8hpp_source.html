77 free_list.reserve(max_elements);

80
81 ~block_allocator() {

87 template <class... Args>
88 locked_ptr<T> allocate(Args &&...args) {
89 auto [index, allocated] = acquire();
90
91 if (!allocated) {
92 large_block[index] = new T(std::forward<Args &&>(args)...);
93 }
94
95 return locked_ptr<T>(large_block[index], index, this->shared_from_this());
96 }
97
98 std::tuple<int_t, bool> acquire() {
99 auto lock = std::lock_guard(mutex);
100
101 if (free_list.empty()) {
102
103 if (large_block.size() == large_block.capacity()) {
104 throw std::length_error("Exceeding capacity of the block_allocator.");
105 }
106
107 large_block.push_back(nullptr);
108 return std::tuple(large_block.size() - 1, false);
109 }
110
111 auto index = free_list.back();
112 free_list.pop_back();
113 return std::tuple(index, true);
114 }
115
116 void release(const locked_ptr<T> &ptr) {
117 auto lock = std::lock_guard(mutex);
118 free_list.push_back(ptr.index);
119 }
120
121private:
122 std::vector<int_t> free_list;
123 std::vector<T *> large_block;
124 std::mutex mutex;
125};
