113 template <class T, int n_dims, template <int N> class Indexing>
114 array<T, n_dims, row_major>
115 array<T, n_dims, Indexing>::load(HierarchicalReader &reader,
116  const std::string &tag) {
117 
118  static_assert(std::is_same<row_major<n_dims>, Indexing<n_dims>>::value,
119  "This has only been implemented for row-major index order.");
120 
121  auto dims = reader.dims(tag);
122  auto shape = shape_t<n_dims>{};
123 
124  // Some types T are stored as an array of fixed size. Such types may require
125  // an additional dimension (at the end of `dims`).
126  LOG_ERR_IF(dims.size() < n_dims, "Dimension mismatch.");
127  for (int_t i = 0; i < n_dims; ++i) {
128  shape[i] = dims[i];
129  }
130 
131  auto arr = array<T, n_dims>(shape, device_type::cpu);
132  load_impl(reader, arr, tag, typename array_save_traits<T>::dispatch_tag{});
133 
134  return arr;
135 }
