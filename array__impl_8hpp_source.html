50template <class T, int n_dims, template <int N> class Indexing>
51array<T, n_dims, Indexing> &array<T, n_dims, Indexing>::operator=(
52 const array_const_view<T, n_dims, Indexing> &other) {
53
54 if (this->shape() != other.shape()) {
55 (*this) = array<T, n_dims, Indexing>(other.shape());
56 }
57
58 // It's pointing to *all* of this array.
59 if (raw_ptr(*this) == raw_ptr(other)) {
60 return *this;
61 }
62
63 std::copy(other.begin(), other.end(), this->begin());
64
65 return *this;
66}

105template <class T, int n_dims>
106void load_impl(HierarchicalReader &reader,
107 array<T, n_dims, row_major> &arr,
108 const std::string &tag,
109 bool_dispatch_tag) {
110
111 using scalar_type = typename array_save_traits<T>::scalar_type;
112 auto datatype = erase_data_type<scalar_type>();
113
114 auto int_arr = array<scalar_type, n_dims, row_major>(arr.shape());
115 reader.read_array(int_arr.raw(), datatype, tag);
116
117 std::copy(int_arr.cbegin(), int_arr.cend(), arr.begin());
118}

120template <class T, int n_dims, template <int N> class Indexing>
121array<T, n_dims, row_major>
122array<T, n_dims, Indexing>::load(HierarchicalReader &reader,
123 const std::string &tag) {
124
125 static_assert(std::is_same<row_major<n_dims>, Indexing<n_dims>>::value,
126 "This has only been implemented for row-major index order.");
127
128 auto dims = reader.dims(tag);
129 auto shape = shape_t<n_dims>{};
130
131 // Some types T are stored as an array of fixed size. Such types may require
132 // an additional dimension (at the end of `dims`).
133 LOG_ERR_IF(dims.size() < n_dims, "Dimension mismatch.");
134 for (int_t i = 0; i < n_dims; ++i) {
135 shape[i] = dims[i];
136 }
137
138 auto arr = array<T, n_dims>(shape, device_type::cpu);
139 load_impl(reader, arr, tag, typename array_save_traits<T>::dispatch_tag{});
140
141 return arr;
142}

145template <class T, int n_dims, template <int N> class Indexing>
146array<T, n_dims, Indexing> uniform(const T &low,
147 const T &high,
148 const shape_t<n_dims> &shape,
149 device_type device) {
150 LOG_ERR_IF(device != device_type::cpu, "Implement first.");
151 static_assert(std::is_same<T, double>::value, "Implement first.");
152
153 auto n = product(shape);
154 auto x = array<T, n_dims, Indexing>(shape, device);
155
156 std::random_device r;
157 auto mt = std::mt19937(r());
158 auto dist = std::uniform_real_distribution<T>(low, high);
159
160 for (int_t i = 0; i < n; ++i) {
161 x[i] = dist(mt);
162 }
163
164 return x;
165}
