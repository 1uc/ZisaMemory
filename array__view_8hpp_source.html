176 template <class T, int n_dims>
177 array_const_view<T, n_dims, row_major>
178 slice(const array_const_view<T, n_dims, row_major> &arr, int_t i0, int_t i1) {
179  auto sub_shape = arr.shape();
180  sub_shape[0] = i1 - i0;
181 
182  int_t offset = i0 * (product(sub_shape) / (i1 - i0));
183  auto ptr = arr.raw() + offset;
184  return {sub_shape, ptr};
185 }
186 }

239 template <class T, int n_dims>
240 void save(HierarchicalWriter &writer,
241  const array_const_view<T, n_dims, row_major> &arr,
242  const std::string &tag,
243  default_dispatch_tag) {
244 
245  T const *const data = arr.raw();
246  const auto &shape = arr.shape();
247 
248  auto data_type = erase_data_type<T>();
249 
250  std::size_t dims[n_dims];
251  for (int_t i = 0; i < n_dims; ++i) {
252  dims[i] = shape(i); // size of (i, j, k) axes
253  }
254 
255  writer.write_array(data, data_type, tag, n_dims, dims);
256 }

258 template <class T, int n_dims>
259 void save(HierarchicalWriter &writer,
260  const array_const_view<T, n_dims, row_major> &arr,
261  const std::string &tag,
262  split_array_dispatch_tag) {
263 
264  using scalar_type = typename array_save_traits<T>::scalar_type;
265  auto data_type = erase_data_type<scalar_type>();
266 
267  constexpr int_t rank = n_dims + 1;
268  std::size_t dims[rank];
269  for (int_t i = 0; i < rank - 1; ++i) {
270  dims[i] = arr.shape(i);

272  dims[rank - 1] = T::size();
