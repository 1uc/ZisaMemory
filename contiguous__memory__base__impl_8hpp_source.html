66template <class T, class Allocator, class Equivalence, class Construction>
67contiguous_memory_base<T, Allocator, Equivalence, Construction> &
68contiguous_memory_base<T, Allocator, Equivalence, Construction>::operator=(
69 const contiguous_memory_base &other) {
70
71 // We're the same.
72 if (this == &other) {
73 return (*this);
74 }
75
76 // The other is empty/null.
77 if (other._raw_data == nullptr) {
78 this->free();
79 return *this;
80 }
81
82 // This one is empty/null.
83 if (this->allocator() == nullptr) {
84 allocate(other.size(), *other.allocator());
85 copy(*this, other);
86 return *this;
87 }
88
89 // Neither is empty/null.
90 if (Equivalence::are_equivalent(*this->allocator(), *other.allocator())) {
91 resize(other.size());
92 } else {
93 this->free();
94 allocate(other.size(), *other.allocator());
95 }
96
97 copy(*this, other);
98 return *this;
99}
