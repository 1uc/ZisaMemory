101template <class T, int n_dims, template <int> class Indexing>
102class array_const_view : public array_view_base<const T, Indexing<n_dims>> {
103
104private:
105 using super = array_view_base<const T, Indexing<n_dims>>;
106 using size_type = typename super::size_type;
107
108public:
109 ANY_DEVICE_INLINE

111 T const *ptr,
112 device_type mem_location = device_type::unknown)
113 : super(shape, ptr, mem_location) {}
114
115 template <class Array, class Shape>
116 array_const_view(const array_base<T, Indexing<n_dims>, Array, Shape> &other)
117 : array_const_view(zisa::shape(other),
118 zisa::raw_ptr(other),
119 zisa::memory_location(other)) {}
120
121 ANY_DEVICE_INLINE
122 array_const_view(const array_view<T, n_dims, Indexing> &other)
123 : super(other.shape(), other.raw(), zisa::memory_location(other)) {}
124
125 array_const_view(const std::vector<T> &v)
126 : array_const_view(shape_t<1>{v.size()}, v.data(), device_type::cpu) {}
127
128 ANY_DEVICE_INLINE const T *raw() const { return this->_ptr; }
129 ANY_DEVICE_INLINE const T &operator[](size_type i) const {
130 return this->_ptr[i];
131 }
132
133 template <class... Ints>
134 ANY_DEVICE_INLINE const T &operator()(Ints... ints) const {
135 auto l = Indexing<n_dims>::linear_index(this->shape(),
136 integer_cast<size_type>(ints)...);
137 return (*this)[l];
138 }
139
140 ANY_DEVICE_INLINE T const *begin() const { return this->_ptr; }
141 ANY_DEVICE_INLINE T const *end() const { return this->_ptr + this->size(); }
142
143 ANY_DEVICE_INLINE T const *cbegin() const { return this->_ptr; }
144 ANY_DEVICE_INLINE T const *cend() const { return this->_ptr + this->size(); }
145};

153template <class T, int n_dims>
154array_const_view<T, n_dims, row_major>
155slice(const array_const_view<T, n_dims, row_major> &arr, int_t i0, int_t i1) {
156 auto sub_shape = arr.shape();
157 sub_shape[0] = i1 - i0;
158
159 int_t offset = i0 * (product(sub_shape) / (i1 - i0));
160 auto ptr = arr.raw() + offset;
161 return {sub_shape, ptr};
162}
163}

221template <class T, int n_dims>
222void save(HierarchicalWriter &writer,
223 const array_const_view<T, n_dims, row_major> &arr,
224 const std::string &tag,
225 default_dispatch_tag) {
226
227 T const *const data = arr.raw();
228 const auto &shape = arr.shape();
229
230 auto data_type = erase_data_type<T>();
231
232 std::size_t dims[n_dims];
233 for (int_t i = 0; i < n_dims; ++i) {
234 dims[i] = shape(i); // size of (i, j, k) axes
235 }
236
237 writer.write_array(data, data_type, tag, n_dims, dims);
238}

240template <class T, int n_dims>
241void save(HierarchicalWriter &writer,
242 const array_const_view<T, n_dims, row_major> &arr,
243 const std::string &tag,
244 split_array_dispatch_tag) {
245
246 using scalar_type = typename array_save_traits<T>::scalar_type;
247 auto data_type = erase_data_type<scalar_type>();
248
249 constexpr int_t rank = n_dims + 1;
250 std::size_t dims[rank];
251 for (int_t i = 0; i < rank - 1; ++i) {
252 dims[i] = arr.shape(i);
253 }
254 dims[rank - 1] = T::size();
255
256 writer.write_array(arr.raw(), data_type, tag, rank, dims);
257}

259template <int n_dims>
260void save(HierarchicalWriter &writer,
261 const array_const_view<bool, n_dims, row_major> &arr,
262 const std::string &tag,
263 bool_dispatch_tag) {
264
265 using scalar_type = typename array_save_traits<bool>::scalar_type;
266 auto int_arr = std::vector<scalar_type>(arr.size());
267 std::copy(arr.cbegin(), arr.cend(), int_arr.begin());
268
269 auto int_arr_view = array_const_view<scalar_type, n_dims, row_major>(
270 arr.shape(), int_arr.data());
271
272 save(writer, int_arr_view, tag);
273}
